module "production_vpc" {

  version = "2.4.0"

  name = "prod-vpc"
  cidr = "10.0.0.0/16"

  enable_nat_gateway = true
  single_nat_gateway = false

  tags = {
    Environment = "production"
    ManagedBy   = "Terraform"
  }
}
